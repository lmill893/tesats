#!/usr/bin/env python3 """ L — Holographic Builder · Local Proxy Server ============================================= This tiny server does two things: 1. Serves hologram_builder.html at http://localhost:5050 2. Proxies AI requests to Anthropic at http://localhost:5050/api HOW TO RUN ---------- 1. Get your Anthropic API key from https://console.anthropic.com 2. Open a terminal in this folder and run: python proxy.py YOUR_API_KEY_HERE Or set the environment variable first: export ANTHROPIC_API_KEY=sk-ant-... python proxy.py 3. Open http://localhost:5050 in your browser 4. Click Begin, allow camera — done! """ import sys, os, json, http.server, urllib.request, urllib.error PORT = 5050 API_URL = "https://api.anthropic.com/v1/messages" API_VERSION = "2023-06-01" # ── Get API key ──────────────────────────────────────────────────────── API_KEY = None if len(sys.argv) > 1: API_KEY = sys.argv[1].strip() if not API_KEY: API_KEY = os.environ.get("ANTHROPIC_API_KEY", "").strip() if not API_KEY: print("\n⚠️ No API key found.") print("Run as: python proxy.py sk-ant-YOUR_KEY_HERE") print("Or set: export ANTHROPIC_API_KEY=sk-ant-YOUR_KEY_HERE\n") API_KEY = input("Paste your Anthropic API key now: ").strip() if not API_KEY: print("No key provided — exiting."); sys.exit(1) print(f"\n✓ API key loaded ({API_KEY[:12]}...)") SCRIPT_DIR = os.path.dirname(os.path.abspath(__file__)) class Handler(http.server.BaseHTTPRequestHandler): def log_message(self, fmt, *args): # Suppress noisy access logs; only show errors if str(args[1]) not in ('200', '304'): print(f" [{args[1]}] {args[0]}") def send_cors(self): self.send_header("Access-Control-Allow-Origin", "*") self.send_header("Access-Control-Allow-Methods", "POST, OPTIONS") self.send_header("Access-Control-Allow-Headers", "Content-Type") def do_OPTIONS(self): self.send_response(200) self.send_cors() self.end_headers() def do_GET(self): path = self.path.split("?")[0] if path in ("/", "/index.html"): path = "/hologram_builder.html" file_path = os.path.join(SCRIPT_DIR, path.lstrip("/")) if not os.path.isfile(file_path): self.send_error(404, f"Not found: {path}"); return ext = os.path.splitext(file_path)[1] mime = {"html":"text/html","js":"application/javascript","css":"text/css", "png":"image/png","ico":"image/x-icon"}.get(ext.lstrip("."), "text/plain") with open(file_path, "rb") as f: data = f.read() self.send_response(200) self.send_header("Content-Type", mime) self.send_header("Content-Length", len(data)) self.send_cors() self.end_headers() self.wfile.write(data) def do_POST(self): if self.path != "/api": self.send_error(404); return length = int(self.headers.get("Content-Length", 0)) body = self.rfile.read(length) req = urllib.request.Request( API_URL, data=body, headers={ "Content-Type": "application/json", "x-api-key": API_KEY, "anthropic-version": API_VERSION, }, method="POST" ) try: with urllib.request.urlopen(req, timeout=60) as resp: result = resp.read() self.send_response(200) self.send_header("Content-Type", "application/json") self.send_cors() self.end_headers() self.wfile.write(result) print(f" [AI] Request OK ({len(result)} bytes)") except urllib.error.HTTPError as e: err_body = e.read() print(f" [AI] Anthropic error {e.code}: {err_body[:200]}") self.send_response(e.code) self.send_header("Content-Type", "application/json") self.send_cors() self.end_headers() self.wfile.write(err_body) except Exception as e: print(f" [AI] Proxy error: {e}") self.send_response(500) self.send_header("Content-Type", "application/json") self.send_cors() self.end_headers() self.wfile.write(json.dumps({"error": str(e)}).encode()) if __name__ == "__main__": server = http.server.HTTPServer(("localhost", PORT), Handler) print(f"\n🌐 Open this in your browser → http://localhost:{PORT}") print(f"📂 Serving from: {SCRIPT_DIR}") print(f" Press Ctrl+C to stop.\n") try: server.serve_forever() except KeyboardInterrupt: print("\nStopped.")
